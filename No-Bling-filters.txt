 //    "path/file"  |       "-"         |  keep "file" (i.e. remove "file" from auto-generated lists)
 //    "path/file"  |       "--"        |  keep all modifiers having asset "file" (reverse-lookup)

 //    "8958"       |       "#"         |  keep all modifiers having item id "number"
